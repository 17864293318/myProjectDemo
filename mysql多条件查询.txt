//  根据reply_count排序，reply_count大于0的排在前面，然后根据last_visit_time倒序排序。
SELECT  q.question_id,            q.userid,            q.city_code,            q.con_type,            q.con_cate_id,            q.con_cate_name,            q.que_cate_id,            q.que_cate_name,            q.title,            q.is_question,            q.reply_count,            q.consult_count,            q.pay_count,            q.consult_status,            q.create_time,            q.start_countdown_time,            q.effective_time,            q.last_visit_time,            q.city_nameFROM (SELECT q.question_id,            q.userid,            q.city_code,            q.con_type,            q.con_cate_id,            q.con_cate_name,            q.que_cate_id,            q.que_cate_name,            q.title,            q.is_question,            q.reply_count,            q.consult_count,            q.pay_count,            q.consult_status,            q.create_time,            q.start_countdown_time,            q.effective_time,            q.last_visit_time,            c.city_nameFROM joy_consult_question q LEFT JOIN joy_city_config c ON q.city_code = c.city_codeWHERE q.userid = 1 and q.reply_count >0ORDER BY q.last_visit_time DESC) qUNION ALLSELECT q.question_id,            q.userid,            q.city_code,            q.con_type,            q.con_cate_id,            q.con_cate_name,            q.que_cate_id,            q.que_cate_name,            q.title,            q.is_question,            q.reply_count,            q.consult_count,            q.pay_count,            q.consult_status,            q.create_time,            q.start_countdown_time,            q.effective_time,            q.last_visit_time,            q.city_nameFROM(SELECT q.question_id,            q.userid,            q.city_code,            q.con_type,            q.con_cate_id,            q.con_cate_name,            q.que_cate_id,            q.que_cate_name,            q.title,            q.is_question,            q.reply_count,            q.consult_count,            q.pay_count,            q.consult_status,            q.create_time,            q.start_countdown_time,            q.effective_time,            q.last_visit_time,            c.city_nameFROM joy_consult_question q LEFT JOIN joy_city_config c ON q.city_code = c.city_codeWHERE q.userid = 1 and q.reply_count =0ORDER BY q.last_visit_time DESC) q
